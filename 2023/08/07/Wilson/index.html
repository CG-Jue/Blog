    from Crypto.Util.number import *
import sympy
import random
import math
from secret import flag

def Ez_gen():
    while True:
        p = getPrime(512)
        if p % 8 == 5:
            break
    g = p - random.randint(1 << 13, 1 << 15)
    q = sympy.nextprime(math.gamma(g+1) % p)
    return p, g, q

p, g, q = Ez_gen()
n = p*q
e = 65537
m = bytes_to_long(flag)
c = pow(m, e, n)

print('p =',p)
print('g =',g)

    知识点

    伽马函数Γ(x)和阶乘x!的关系式,n为正整数
    gamma(n)=(n-1)!
    #威尔逊定理
    当且仅当p为素数时：(p-1)! ≡ -1 (mod p)
    推论：(p-2)! ≡ 1 (mod p)
    #欧拉函数若n是质数p的k次幂，phi = p**k - p**(k-1)

    分析

    很明显我们要求q

    也就是要求gamma(g+1)%p,

    已知gamma(g+1)%p===g!%p
    根据威尔逊定理可知：(p-2)%p=1
    令x=(p-2)！/g！，y=g!,所以(x*y)%p=1
    我们要求y%p,y是x关于p的逆元，所以求x%p的逆元就可
